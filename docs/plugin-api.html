Plugin API
接口文档

为 Chrome 插件提供 Google OAuth 登录、用户配额管理及 AI 对话能力（基于 OpenRouter 免费模型）。

BASE URL https://ai.zhoudw.vip/api
认证接口
POST /plugin/auth/google/login
插件通过 Google Identity Services 获取 credential（id_token）后，POST 到此接口完成登录或自动注册。 成功后返回 Sa-Token，后续请求通过 Authorization: Bearer <token> 携带。
Request Body (JSON)
字段类型必填说明
idToken string 必填 Google Identity Services 返回的 credential（JWT id_token）
Request
// 插件端 Google 登录触发
google.accounts.id.initialize({
  client_id: 'YOUR_GOOGLE_CLIENT_ID',
  callback: async (response) => {
    const res = await fetch('https://ai.zhoudw.vip/api/plugin/auth/google/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ idToken: response.credential })
    });
    const data = await res.json();
    // data.data.token → 保存到 chrome.storage.local
  }
});

AI 对话接口
POST /plugin/ai/chat/completions 需登录
OpenAI 兼容格式的对话补全接口。支持 stream: true 开启 SSE 流式输出。 遇到上游 429 限速时自动轮询候选模型重试（非流式），流式使用 openrouter/free 自动路由。
Request Body (JSON)
字段类型必填说明
messages array 必填 消息列表，格式见下方示例
stream boolean 可选 是否启用 SSE 流式输出，默认 false
messages[].role 支持值
role说明
system系统提示词
user用户消息
assistantAI 历史消息（多轮对话）

流式示例（stream: true）
流式模式使用 openrouter/free 自动路由，由 OpenRouter 服务端自动选择可用的免费模型， 返回标准 SSE 格式（OpenAI 兼容），最后发送 data: [DONE] 标记结束。
Request + SSE 消费
async function streamChat(token, messages, onChunk) {
  const res = await fetch('https://ai.zhoudw.vip/api/plugin/ai/chat/completions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`
    },
    body: JSON.stringify({ messages, stream: true })
  });

  const reader = res.body.getReader();
  const decoder = new TextDecoder();
  let buf = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    buf += decoder.decode(value, { stream: true });
    const lines = buf.split('\n');
    buf = lines.pop();                 // 保留未完整的行

    for (const line of lines) {
      if (!line.startsWith('data: ')) continue;
      const raw = line.slice(6).trim();
      if (raw === '[DONE]') return;     // 流结束
      try {
        const chunk = JSON.parse(raw);
        const content = chunk.choices?.[0]?.delta?.content;
        if (content) onChunk(content);
      } catch {}
    }
  }
}

// 使用示例
let output = '';
await streamChat(token, [
  { role: 'user', content: '用三句话总结这篇文档...' }
], (chunk) => {
  output += chunk;
  updateUI(output);   // 实时更新插件 UI
});

错误码
code场景说明
200 所有成功响应 操作成功
401 未携带 token 或 token 已过期 请重新登录获取新 token
429 今日配额已耗尽 免费用户每日 20 次 / 10 万 tokens，次日重置
500 全部候选模型均不可用 返回 msg: "AI 服务繁忙，请稍后重试"
流式连接中断（网络错误）时，插件端应捕获异常并提示用户重试，服务端不做流式 fallback。

plugin/ai-client.js
const BASE = 'https://ai.zhoudw.vip/api';

/** 从 chrome.storage 取 token */
async function getToken() {
  const { pluginToken } = await chrome.storage.local.get('pluginToken');
  return pluginToken;
}

/** Google 登录 */
export async function googleLogin(idToken) {
  const res = await fetch(`${BASE}/plugin/auth/google/login`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ idToken })
  });
  const { data } = await res.json();
  await chrome.storage.local.set({ pluginToken: data.token });
  return data;
}

/** 刷新配额显示 */
export async function refreshQuota() {
  const token = await getToken();
  const res = await fetch(`${BASE}/plugin/auth/quota`, {
    headers: { 'Authorization': `Bearer ${token}` }
  });
  const { data } = await res.json();
  return data; // { remainingRequests, requestLimit, remainingTokens, ... }
}

/** 流式 AI 对话 */
export async function streamChat(messages, onChunk, onDone) {
  const token = await getToken();
  const res = await fetch(`${BASE}/plugin/ai/chat/completions`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`
    },
    body: JSON.stringify({ messages, stream: true })
  });

  if (!res.ok) {
    const err = await res.json();
    throw new Error(err.msg || '请求失败');
  }

  const reader = res.body.getReader();
  const decoder = new TextDecoder();
  let buf = '', fullText = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    buf += decoder.decode(value, { stream: true });
    const lines = buf.split('\n');
    buf = lines.pop();

    for (const line of lines) {
      if (!line.startsWith('data: ')) continue;
      const raw = line.slice(6).trim();
      if (raw === '[DONE]') { onDone?.(fullText); return; }
      try {
        const chunk = JSON.parse(raw);
        const content = chunk.choices?.[0]?.delta?.content ?? '';
        if (content) { fullText += content; onChunk(content); }
      } catch {}
    }
  }
}

// ---- 调用示例 ----
let output = '';
await streamChat(
  [{ role: 'user', content: '总结这段文档：...' }],
  (chunk) => { output += chunk; renderOutput(output); },
  async () => {
    const quota = await refreshQuota();
    updateQuotaBadge(quota.remainingRequests); // 对话后刷新次数
  }
);
